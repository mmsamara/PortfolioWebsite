                                    //Code sample from the third exercise (Find the longest word(s) given a list of words)
                                    public string BuildFindLongestWordsOutput(string word1, string word2, string word3, string word4, string word5)
                                    {
                                        var wordArray = new string[] { word2, word3, word4, word5 };
                                        var output = new StringBuilder();
                                        var longestWord = word1;
                                        var longestWords = word1;
                                        foreach (var word in wordArray)
                                        {
                                            if (!string.IsNullOrEmpty(word))
                                            {
                                                if (word.Length > longestWord.Length)
                                                {
                                                    longestWord = word;
                                                    longestWords = longestWord;
                                                }
                                                else if (word.Length == longestWord.Length)
                                                {
                                                    longestWords += "," + word;
                                                }
                                            }
                                        }

                                        var longestWordsArray = longestWords.Split(',').Distinct();

                                        output.AppendLine();
                                        output.AppendLine("I found " + longestWordsArray.Count() + " longest word(s). The words are: ");
                                        foreach (var longWord in longestWordsArray)
                                        {
                                            output.AppendLine(longWord);
                                        }
                                        return output.ToString();
                                    }
